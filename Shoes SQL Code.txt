create table customers(
customer_id NUMERIC GENERATED ALWAYS AS IDENTITY(START WITH 1 INCREMENT BY 1) NOT NULL,
username varchar2(255) unique not null,
pass varchar2(255) not null,
gender  VARCHAR(6) NOT NULL CHECK (Gender IN ('Male', 'Female')),
constraint cu_id_pk primary key (customer_id)
);

create table shoes(
shoe_id NUMERIC GENERATED ALWAYS AS IDENTITY(START WITH 1 INCREMENT BY 1) NOT NULL,
Category_id number not null,
shoe_name  Varchar2(255) not null,
size_id numeric not null, 
price number(15,2),
gender VARCHAR(6) NOT NULL CHECK (Gender IN ('Male', 'Female')),
quantity_available numeric not null,
constraint sh_id_pk primary key (shoe_id),
constraint ca_fk foreign key (category_id) REFERENCES category_table (category_id)
);


create table size_table(
size_id  NUMERIC GENERATED ALWAYS AS IDENTITY(START WITH 1 INCREMENT BY 1) NOT NULl,
shoe_id numeric  not null,
shoe_size varchar (255) not null, 
CONSTRAINT s_id_pk primary key (size_id),

constraint sh_fk foreign key (shoe_id) REFERENCES shoes (shoe_id)
);

create table category_table(
category_id  NUMERIC GENERATED ALWAYS AS IDENTITY(START WITH 1 INCREMENT BY 1) NOT NULl,
category_name varchar2(255) unique not null,
constraint ca_id_pk primary key (category_id)
);



create table order_table(
order_id  NUMERIC GENERATED ALWAYS AS IDENTITY(START WITH 1 INCREMENT BY 1) NOT NULl,
shoe_id numeric not null,
customer_id numeric not null,
size_id numeric not null,
category_id numeric not null,
quantity numeric  not null,
total numeric not null,
constraint si_fk foreign key (shoe_id) REFERENCES shoes (shoe_id),
constraint cu_fk foreign key (customer_id) REFERENCES customers (customer_id),
constraint size_fk foreign key (size_id) REFERENCES size_table (size_id),
constraint ca_fk foreign key (category_id) REFERENCES category_table (category_id),
CONSTRAINT order_id_pk primary key (order_id)
);
